main:
	li t0,0			# 1st Number
    li t1,1			# 2nd Number
    li t3,10		# Limit
    li t4,1			# Counter
    jalr ra,zero,20
swapping:
	# Result = 1st Number + 2nd Number
    add t2,t0,t1
    # Updating 1st Number
    sw t1,0x0(x0)
    lw t0,0x0(x0)
    # Updating 2nd Number
    sw t2,0x0(x0)
    lw t1,0x0(x0)
    # Incrementing Counter
    addi t4,t4,1
    beq t3,t4,end
    ret
end:
